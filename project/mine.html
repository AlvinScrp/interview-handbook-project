56    const date = originDate ? dayjs(originDate) : dayjs()
57    const days: DayItem[] = []
58    const len = 42
59
60    // 当前月
61    const currDays = date.daysInMonth()
62    for (let index = 1; index <= currDays; index++) {
63      days.push({
64        date: index,
65        month: date.month() + 1,
66        year: date.year(),
67        isSelected: this.selectedDays.some(item => date.date(index).isSame(item))
68      })
69    }
70    // 上个月
71    const prevMonth = date.date(0)
72    const prevMonthLastDate = prevMonth.date()
73    const prevDays = prevMonth.day()
74    if (prevDays < 6) {
75      for (let index = 0; index <= prevDays; index++) {
76        days.unshift({
77          date: prevMonthLastDate - index,
78          month: prevMonth.month() + 1,
79          year: prevMonth.year()
80        })
81      }
82    }
83    // 下个月
84    const nextMonth = date.date(currDays + 1)
85    const start = days.length
86    for (let index = 1; index <= len - start; index++) {
87      days.push({
88        date: index,
89        month: nextMonth.month() + 1,
90        year: nextMonth.year()
91      })
92    }
93
94    return days
95  }

98    return dayjs(this.currentDate).month() === month - 1
